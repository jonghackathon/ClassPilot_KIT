# 05. 인프라 및 공통 패턴

> 현재 프로젝트 기준 공통 import는 `src/lib/auth.ts`의 `auth()`와 `src/lib/db.ts`의 `prisma`를 사용한다. 아래 예제는 모두 이 구조를 기준으로 해석한다.

## 1. SWR 패턴

### Fetcher 설정

```typescript
// lib/fetcher.ts
export const fetcher = async (url: string) => {
  const res = await fetch(url);
  if (!res.ok) {
    const error = new Error("API 요청 실패");
    const body = await res.json().catch(() => ({}));
    (error as any).status = res.status;
    (error as any).info = body;
    throw error;
  }
  return res.json();
};
```

### SWR 키 규칙

```
/api/[도메인]                    → 목록 조회
/api/[도메인]?classId=xxx        → 필터링된 목록
/api/[도메인]/[id]               → 단건 조회
/api/[도메인]?page=1&limit=20   → 페이지네이션
```

예시:
- `/api/attendance?classId=cls_01&date=2026-04-08` → 특정 반의 오늘 출석
- `/api/assignments?classId=cls_01&status=pending` → 미제출 과제 목록
- `/api/users?role=STUDENT&classId=cls_01` → 특정 반 학생 목록

### Optimistic Update 패턴

```typescript
// hooks/useAttendance.ts
import useSWR, { mutate } from "swr";

export function useAttendance(classId: string, date: string) {
  const key = `/api/attendance?classId=${classId}&date=${date}`;
  const { data, error, isLoading } = useSWR(key, fetcher);

  const markAttendance = async (studentId: string, status: "PRESENT" | "ABSENT" | "LATE") => {
    // 1. 현재 데이터 백업
    const previousData = data;

    // 2. Optimistic update — UI 즉시 반영
    mutate(key, (current: any) => ({
      ...current,
      data: current.data.map((item: any) =>
        item.studentId === studentId ? { ...item, status } : item
      ),
    }), false); // revalidate: false → 서버 요청 전에 UI 먼저 갱신

    try {
      // 3. 서버 요청
      await fetch("/api/attendance", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ studentId, classId, date, status }),
      });

      // 4. 성공 시 서버 데이터로 재검증
      mutate(key);
    } catch (error) {
      // 5. 실패 시 롤백
      mutate(key, previousData, false);
      throw error;
    }
  };

  return { data, error, isLoading, markAttendance };
}
```

## 2. API Route Handler 패턴

### 표준 응답 포맷

```typescript
// lib/api-response.ts
import { NextResponse } from "next/server";

type ApiResponse<T = any> = {
  success: boolean;
  data?: T;
  error?: string;
};

export function successResponse<T>(data: T, status = 200) {
  return NextResponse.json({ success: true, data } satisfies ApiResponse<T>, { status });
}

export function errorResponse(error: string, status = 400) {
  return NextResponse.json({ success: false, error } satisfies ApiResponse, { status });
}
```

### GET 핸들러 템플릿

```typescript
// src/app/api/[도메인]/route.ts
import { auth } from "@/lib/auth";
import { prisma } from "@/lib/db";
import { successResponse, errorResponse } from "@/lib/api-response";

export async function GET(req: Request) {
  // 1. 인증 확인
  const session = await auth();
  if (!session) return errorResponse("인증이 필요합니다", 401);

  // 2. 역할 확인 (필요 시)
  if (session.user.role !== "ADMIN" && session.user.role !== "TEACHER") {
    return errorResponse("권한이 없습니다", 403);
  }

  // 3. 쿼리 파라미터 추출
  const { searchParams } = new URL(req.url);
  const classId = searchParams.get("classId");

  try {
    // 4. Prisma 쿼리
    const items = await prisma.attendance.findMany({
      where: { ...(classId && { classId }) },
      include: { student: { select: { name: true } } },
      orderBy: { createdAt: "desc" },
    });

    // 5. 성공 응답
    return successResponse(items);
  } catch (error) {
    console.error("[GET /api/attendance]", error);
    return errorResponse("데이터 조회 실패", 500);
  }
}
```

### POST 핸들러 템플릿

```typescript
export async function POST(req: Request) {
  const session = await auth();
  if (!session) return errorResponse("인증이 필요합니다", 401);

  try {
    // 1. Request body 파싱 + 유효성 검사
    const body = await req.json();
    const parsed = attendanceSchema.safeParse(body);
    if (!parsed.success) {
      return errorResponse(parsed.error.errors[0].message, 422);
    }

    // 2. 비즈니스 로직 (중복 체크 등)
    const existing = await prisma.attendance.findFirst({
      where: {
        studentId: parsed.data.studentId,
        date: parsed.data.date,
      },
    });
    if (existing) return errorResponse("이미 출석 체크됨", 409);

    // 3. 생성
    const created = await prisma.attendance.create({
      data: parsed.data,
    });

    return successResponse(created, 201);
  } catch (error) {
    console.error("[POST /api/attendance]", error);
    return errorResponse("생성 실패", 500);
  }
}
```

### PATCH 핸들러 템플릿

```typescript
// src/app/api/[도메인]/[id]/route.ts
export async function PATCH(
  req: Request,
  { params }: { params: { id: string } }
) {
  const session = await auth();
  if (!session) return errorResponse("인증이 필요합니다", 401);

  try {
    const body = await req.json();
    const parsed = attendanceUpdateSchema.safeParse(body);
    if (!parsed.success) {
      return errorResponse(parsed.error.errors[0].message, 422);
    }

    const updated = await prisma.attendance.update({
      where: { id: params.id },
      data: parsed.data,
    });

    return successResponse(updated);
  } catch (error: any) {
    if (error.code === "P2025") return errorResponse("항목을 찾을 수 없습니다", 404);
    console.error("[PATCH /api/attendance/[id]]", error);
    return errorResponse("수정 실패", 500);
  }
}
```

### 인증 + 역할 미들웨어 패턴

```typescript
// src/lib/with-auth.ts
import { auth } from "@/lib/auth";
import { errorResponse } from "@/lib/api-response";

type Role = "ADMIN" | "TEACHER" | "STUDENT";

export async function withAuth(roles?: Role[]) {
  const session = await auth();

  if (!session) {
    return { error: errorResponse("인증이 필요합니다", 401), session: null };
  }

  if (roles && !roles.includes(session.user.role as Role)) {
    return { error: errorResponse("권한이 없습니다", 403), session: null };
  }

  return { error: null, session };
}

// 사용 예시:
export async function GET(req: Request) {
  const { error, session } = await withAuth(["ADMIN", "TEACHER"]);
  if (error) return error;

  // session은 null이 아님이 보장됨
  // ... 비즈니스 로직
}
```

## 4. 인증 완성 계획

### 완성 계획

#### 4-1. PrismaAdapter 활성화

> 현재 프로젝트는 `authOptions` 단독 export가 아니라 `src/lib/auth.ts`에서 `NextAuth(...)` 결과를 구조분해하는 패턴을 사용한다. 아래 예제의 핵심은 `PrismaAdapter(prisma)`와 `bcrypt.compare()` 흐름이며, 실제 구현 형태는 `export const { handlers, auth, signIn, signOut } = NextAuth(...)`에 맞춘다.

```typescript
// src/lib/auth.ts
import NextAuth from "next-auth";
import CredentialsProvider from "next-auth/providers/credentials";
import { PrismaAdapter } from "@auth/prisma-adapter";
import { prisma } from "@/lib/db";
import bcrypt from "bcryptjs";

export const { handlers, auth, signIn, signOut } = NextAuth({
  adapter: PrismaAdapter(prisma),
  providers: [
    CredentialsProvider({
      name: "credentials",
      credentials: {
        email: { label: "Email", type: "email" },
        password: { label: "Password", type: "password" },
      },
      async authorize(credentials) {
        if (!credentials?.email || !credentials?.password) return null;

        const user = await prisma.user.findUnique({
          where: { email: credentials.email },
        });

        if (!user || !user.password) return null;

        const isValid = await bcrypt.compare(credentials.password, user.password);
        if (!isValid) return null;

        return {
          id: user.id,
          email: user.email,
          name: user.name,
          role: user.role,
        };
      },
    }),
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.role = user.role;
        token.id = user.id;
      }
      return token;
    },
    async session({ session, token }) {
      if (session.user) {
        session.user.role = token.role as string;
        session.user.id = token.id as string;
      }
      return session;
    },
  },
  session: { strategy: "jwt" },
  pages: { signIn: "/login" },
});
```

```typescript
// middleware.ts
import { withAuth } from "next-auth/middleware";

export default withAuth(
  function middleware(req) {
    const { pathname } = req.nextUrl;
    const role = req.nextauth.token?.role;

    if (pathname.startsWith("/admin") && role !== "ADMIN") {
      return NextResponse.redirect(new URL("/", req.url));
    }
    if (pathname.startsWith("/teacher") && role !== "TEACHER" && role !== "ADMIN") {
      return NextResponse.redirect(new URL("/", req.url));
    }
    if (pathname.startsWith("/student") && role !== "STUDENT" && role !== "ADMIN") {
      return NextResponse.redirect(new URL("/", req.url));
    }
  },
  { callbacks: { authorized: ({ token }) => !!token } }
);

export const config = {
  matcher: ["/admin/:path*", "/teacher/:path*", "/student/:path*"],
};
```

## 5. Zustand 스토어

### 사용 기준

| 용도 | 도구 | 이유 |
|------|------|------|
| 서버 데이터 (출석, 과제, 학생 목록 등) | **SWR** | 캐시 + 재검증 + 자동 갱신 |
| 페이지 간 유지되는 UI 상태 | **Zustand** | 라우트 전환 후에도 유지 |
| 단일 컴포넌트 내 로컬 상태 | **useState** | 가장 단순 |

### Zustand 적용 대상

```typescript
// store/useAppStore.ts
import { create } from "zustand";
import { persist } from "zustand/middleware";

interface AppState {
  // 현재 선택된 반 (여러 페이지에서 참조)
  currentClassId: string | null;
  setCurrentClassId: (id: string | null) => void;

  // 사이드바 열림/닫힘
  sidebarOpen: boolean;
  toggleSidebar: () => void;

  // AI Copilot 패널 열림/닫힘
  copilotOpen: boolean;
  toggleCopilot: () => void;
}

export const useAppStore = create<AppState>()(
  persist(
    (set) => ({
      currentClassId: null,
      setCurrentClassId: (id) => set({ currentClassId: id }),

      sidebarOpen: true,
      toggleSidebar: () => set((s) => ({ sidebarOpen: !s.sidebarOpen })),

      copilotOpen: false,
      toggleCopilot: () => set((s) => ({ copilotOpen: !s.copilotOpen })),
    }),
    { name: "classpilot-app-store" } // localStorage 키
  )
);
```

### SWR vs Zustand 판단 흐름

```
데이터가 서버에서 오는가?
  ├─ Yes → SWR (useSWR)
  │        예: 학생 목록, 출석 기록, 과제 목록
  └─ No → 여러 페이지에서 공유하는가?
           ├─ Yes → Zustand
           │        예: 선택된 반, 사이드바 상태
           └─ No → useState
                    예: 모달 열림, 폼 입력값
```

```typescript
// hooks/useAttendance.ts
import useSWR from "swr";
import { fetcher } from "@/lib/fetcher";

export function useAttendance(classId?: string, date?: string) {
  const params = new URLSearchParams();
  if (classId) params.set("classId", classId);
  if (date) params.set("date", date);

  const key = classId ? `/api/attendance?${params.toString()}` : null;
  const { data, error, isLoading, mutate } = useSWR(key, fetcher);

  const markAttendance = async (studentId: string, status: string) => {

      body: JSON.stringify({ studentId, classId, date, status }),
    });
    if (!res.ok) throw new Error("출석 체크 실패");
    mutate();
  };

  return {
    attendance: data?.data ?? [],
    error,
    isLoading,
    markAttendance,
    mutate,
  };
}
```

## 7. 에러 처리

### HTTP 상태 코드 규칙

| 코드 | 의미 | 사용 상황 |
|------|------|----------|
| `200` | 성공 | GET, PATCH 성공 |
| `201` | 생성됨 | POST 성공 |
| `204` | 내용 없음 | DELETE 성공 |
| `400` | 잘못된 요청 | 파라미터 누락 |
| `401` | 인증 필요 | 로그인 안 됨 |
| `403` | 권한 없음 | 역할 부족 |
| `404` | 찾을 수 없음 | 존재하지 않는 리소스 |
| `409` | 충돌 | 중복 데이터 |
| `422` | 유효성 실패 | Zod 검증 실패 |
| `500` | 서버 오류 | 예외 처리 안 된 에러 |

```typescript
const supabase = createClient(

// 버킷 구조:
// - assignments/  → 과제 첨부파일
// - recordings/   → 수업 녹음 파일
// - profiles/     → 프로필 이미지

export async function uploadFile(
  bucket: string,
  path: string,
  file: Buffer,
  contentType: string
) {
  const { data, error } = await supabase.storage
    .from(bucket)
    .upload(path, file, {
      contentType,
      upsert: true,
    });

  if (error) throw error;

  const { data: urlData } = supabase.storage
    .from(bucket)
    .getPublicUrl(data.path);

  return urlData.publicUrl;
}

export async function deleteFile(bucket: string, path: string) {
  const { error } = await supabase.storage.from(bucket).remove([path]);
  if (error) throw error;
}
```

### 과제 이미지 업로드 API

```typescript
// app/api/upload/route.ts
import { uploadFile } from "@/lib/supabase-storage";
import { withAuth } from "@/lib/with-auth";
import { successResponse, errorResponse } from "@/lib/api-response";

export async function POST(req: Request) {
  const { error, session } = await withAuth();
  if (error) return error;

  const formData = await req.formData();
  const file = formData.get("file") as File;
  const bucket = formData.get("bucket") as string || "assignments";

  if (!file) return errorResponse("파일이 없습니다", 400);

  // 파일 크기 제한 (10MB)
  if (file.size > 10 * 1024 * 1024) {
    return errorResponse("파일 크기는 10MB 이하여야 합니다", 400);
  }

  // 허용 타입
  const allowedTypes = ["image/jpeg", "image/png", "image/webp", "audio/webm", "audio/mp4"];
  if (!allowedTypes.includes(file.type)) {
    return errorResponse("허용되지 않는 파일 형식입니다", 400);
  }

  const buffer = Buffer.from(await file.arrayBuffer());
  const ext = file.name.split(".").pop();
  const path = `${session!.user.id}/${Date.now()}.${ext}`;

  const url = await uploadFile(bucket, path, buffer, file.type);
  return successResponse({ url }, 201);
}
```

## 9. SSE 스트리밍

### 서버측 (AI Copilot)

```typescript
// app/api/ai/copilot/route.ts
import Anthropic from "@anthropic-ai/sdk";
import { withAuth } from "@/lib/with-auth";

export async function POST(req: Request) {
  const { error, session } = await withAuth();
  if (error) return error;

  const { message, history } = await req.json();

  const anthropic = new Anthropic();

  // SSE 스트림 생성
  const encoder = new TextEncoder();
  const stream = new ReadableStream({
    async start(controller) {
      try {
        const response = await anthropic.messages.stream({
          model: "claude-sonnet-4-20250514",
          max_tokens: 2048,
          system: "당신은 학원 운영을 돕는 AI 어시스턴트입니다. 한국어로 답변합니다.",
          messages: history.concat([{ role: "user", content: message }]),
        });

        for await (const event of response) {
          if (event.type === "content_block_delta" && event.delta.type === "text_delta") {
            const data = JSON.stringify({ text: event.delta.text });
            controller.enqueue(encoder.encode(`data: ${data}\n\n`));
          }
        }

        controller.enqueue(encoder.encode("data: [DONE]\n\n"));
        controller.close();
      } catch (err) {
        const data = JSON.stringify({ error: "AI 응답 생성 실패" });
        controller.enqueue(encoder.encode(`data: ${data}\n\n`));
        controller.close();
      }
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
      Connection: "keep-alive",
    },
  });
}
```

### 클라이언트측 (EventSource 관리)

```typescript
// hooks/useCopilot.ts
import { useState, useCallback, useRef } from "react";

interface Message {
  role: "user" | "assistant";
  content: string;
}

export function useCopilot() {
  const [messages, setMessages] = useState<Message[]>([]);
  const [isStreaming, setIsStreaming] = useState(false);
  const abortRef = useRef<AbortController | null>(null);

  const sendMessage = useCallback(async (userMessage: string) => {
    setMessages((prev) => [...prev, { role: "user", content: userMessage }]);
    setIsStreaming(true);

    // 이전 스트림 취소
    abortRef.current?.abort();
    abortRef.current = new AbortController();

    let assistantContent = "";
    setMessages((prev) => [...prev, { role: "assistant", content: "" }]);

    try {
      const res = await fetch("/api/ai/copilot", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          message: userMessage,
          history: messages,
        }),
        signal: abortRef.current.signal,
      });

      const reader = res.body!.getReader();
      const decoder = new TextDecoder();

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        const chunk = decoder.decode(value);
        const lines = chunk.split("\n").filter((line) => line.startsWith("data: "));

        for (const line of lines) {
          const data = line.slice(6); // "data: " 제거
          if (data === "[DONE]") break;

          const parsed = JSON.parse(data);
          if (parsed.text) {
            assistantContent += parsed.text;
            setMessages((prev) => {
              const updated = [...prev];
              updated[updated.length - 1] = {
                role: "assistant",
                content: assistantContent,
              };
              return updated;
            });
          }
        }
      }
    } catch (err: any) {
      if (err.name !== "AbortError") {
        console.error("Copilot 스트리밍 실패:", err);
      }
    } finally {
      setIsStreaming(false);
    }
  }, [messages]);

  const stopStreaming = useCallback(() => {
    abortRef.current?.abort();
    setIsStreaming(false);
  }, []);

  return { messages, isStreaming, sendMessage, stopStreaming };
}
```

### 재연결 전략

```typescript
// EventSource 연결이 끊겼을 때 자동 재연결
// fetch + ReadableStream 방식에서는 에러 시 재시도 로직으로 대체

const MAX_RETRIES = 3;
const RETRY_DELAY = 1000; // 1초

async function fetchWithRetry(url: string, options: RequestInit, retries = 0): Promise<Response> {
  try {
    return await fetch(url, options);
  } catch (error) {
    if (retries < MAX_RETRIES) {
      await new Promise((r) => setTimeout(r, RETRY_DELAY * (retries + 1)));
      return fetchWithRetry(url, options, retries + 1);
    }
    throw error;
  }
}
```
