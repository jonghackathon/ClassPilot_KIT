### 4.3 middleware.ts — 라우팅 가드

```typescript
// middleware.ts
import { withAuth } from 'next-auth/middleware'
import { NextResponse } from 'next/server'

export default withAuth(
  function middleware(req) {
    const token = req.nextauth.token
    const path = req.nextUrl.pathname

    // ADMIN 전용 경로
    if (path.startsWith('/dashboard') && token?.role !== 'ADMIN') {
      if (token?.role === 'TEACHER') return NextResponse.redirect(new URL('/teacher/dashboard', req.url))
      if (token?.role === 'STUDENT') return NextResponse.redirect(new URL('/home', req.url))
    }

    // TEACHER 전용 경로
    if (path.startsWith('/teacher') && token?.role !== 'TEACHER') {
      return NextResponse.redirect(new URL('/login', req.url))
    }

    // STUDENT 전용 경로
    if (path.startsWith('/student') && token?.role !== 'STUDENT') {
      return NextResponse.redirect(new URL('/login', req.url))
    }
  },
  {
    callbacks: {
      authorized: ({ token }) => !!token
    }
  }
)

export const config = {
  matcher: ['/((?!api|_next|login).*)']
}
```

### 5.1 수업 코파일럿 (SSE 스트리밍)

```
강사 브라우저              Route Handler              Claude API
     │                         │                         │
     │  POST /api/ai/copilot   │                         │
     │  /ask                   │                         │
     │  { sessionId, question }│                         │
     ├────────────────────────►│                         │
     │                         │  세션 컨텍스트 로드       │
     │                         │  (수업 주제, 레벨 구성)   │
     │                         │                         │
     │                         │  anthropic.messages     │
     │                         │  .create({ stream:true})│
     │                         ├────────────────────────►│
     │                         │                         │
     │  text/event-stream      │  chunk 스트리밍          │
     │  ◄─────────────────────►│◄────────────────────────┤
     │                         │                         │
     │  data: { type:'chunk' } │                         │
     │  data: { type:'chunk' } │                         │
     │  data: { type:'done'  } │                         │
     │                         │  DB 저장                 │
     │                         │  (CopilotQuestion)      │
     │                         ├──────────────────────── │
     │  카드 4개 렌더링 완료      │                         │
```

**useCopilot 훅 구현**

```typescript
// hooks/useCopilot.ts
export function useCopilot(sessionId: string) {
  const [cards, setCards] = useState<CopilotCards | null>(null)
  const [isLoading, setIsLoading] = useState(false)

  const ask = async (question: string) => {
    setIsLoading(true)
    setCards(null)

    const res = await fetch('/api/ai/copilot/ask', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ sessionId, question })
    })

    const reader = res.body!.getReader()
    const decoder = new TextDecoder()

    while (true) {
      const { done, value } = await reader.read()
      if (done) break

      const lines = decoder.decode(value).split('\n')
      for (const line of lines) {
        if (!line.startsWith('data: ')) continue
        const data = JSON.parse(line.slice(6))

        if (data.type === 'done') {
          setCards(data.response)
          setIsLoading(false)
        }
      }
    }
  }

  return { cards, isLoading, ask }
}
```

### 9.6 Route Handler 기본 패턴

```typescript
// app/api/attendance/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { getServerSession } from 'next-auth'
import { authOptions } from '@/lib/auth'
import { prisma } from '@/lib/prisma'
import { z } from 'zod'

const schema = z.object({
  lessonId: z.string(),
  studentId: z.string(),
  status: z.enum(['PRESENT', 'LATE', 'EARLY_LEAVE', 'ABSENT']),
  note: z.string().optional()
})

export async function POST(req: NextRequest) {
  // 1. 인증 확인
  const session = await getServerSession(authOptions)
  if (!session) {
    return NextResponse.json(
      { success: false, error: { code: 'UNAUTHORIZED', message: '로그인이 필요합니다.' } },
      { status: 401 }
    )
  }

  // 2. 권한 확인
  if (!['ADMIN', 'TEACHER'].includes(session.user.role)) {
    return NextResponse.json(
      { success: false, error: { code: 'FORBIDDEN', message: '권한이 없습니다.' } },
      { status: 403 }
    )
  }

  // 3. 유효성 검사
  const body = await req.json()
  const parsed = schema.safeParse(body)
  if (!parsed.success) {
    return NextResponse.json(
      { success: false, error: { code: 'VALIDATION_ERROR', message: '입력값을 확인해주세요.', fields: parsed.error.errors } },
      { status: 400 }
    )
  }

  // 4. 비즈니스 로직
  try {
    const attendance = await prisma.attendance.upsert({
      where: {
        lessonId_studentId: {
          lessonId: parsed.data.lessonId,
          studentId: parsed.data.studentId
        }
      },
      update: { status: parsed.data.status, note: parsed.data.note },
      create: {
        ...parsed.data,
        recordedBy: session.user.id
      }
    })

    return NextResponse.json({ success: true, data: attendance }, { status: 201 })
  } catch (error) {
    return NextResponse.json(
      { success: false, error: { code: 'INTERNAL_ERROR', message: '서버 오류가 발생했습니다.' } },
      { status: 500 }
    )
  }
}
```
